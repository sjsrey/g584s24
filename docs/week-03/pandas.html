Data Cleaning

import numpy as np
# create a dataframe with missing values

data = {
    "Name": ['Alice', 'Bob', None, 'David', 'Eva'], 
    "Age" : [ 25, None, 35, 40, 'Unknown' ],
    "Salary": [50000,  60000, 55000, None, 45000],
    "City": ['New York', 'Los Angeles', np.nan, 'Houston',
             'Phoenix' ]
}
df = pd.DataFrame(data)
